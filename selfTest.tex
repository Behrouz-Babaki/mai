\subsection*{Mathematics}

\begin{question}
What is the result of $\sum_{i=2}^6 i$?
\begin{solution}
\begin{hint}
Recall that $\sum_{i=2}^6 i = 2 + 3 + 4 + 5 + 6$.
\end{hint}
\begin{hint}
Recall that $\sum_{i=1}^n i = \frac{n(n+1)}{2}$.
\end{hint}
The sum is equal to \answer{20}.
\end{solution}
\end{question}
