\subsection*{Mathematics}

\begin{question}
What is the result of $\sum_{i=2}^6 i$?
\begin{solution}
The sum is equal to \answer{20}.
\end{solution}
This sum could be computed via
\[
\sum_{i=2}^6 i = 2 + 3 + 4 + 5 + 6,
\]
or by using the formula
\[
\sum_{i=1}^n i = \frac{n(n+1)}{2}.
\]
\end{question}
